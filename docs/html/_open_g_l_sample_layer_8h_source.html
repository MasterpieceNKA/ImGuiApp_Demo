169 void add_shader(GLuint program, const char* shader_code, GLenum type)
170 {
171 GLuint current_shader = glCreateShader(type);
172
173 const GLchar* code[1];
174 code[0] = shader_code;
175
176 GLint code_length[1];
177 code_length[0] = strlen(shader_code);
178
179 glShaderSource(current_shader, 1, code, code_length);
180 glCompileShader(current_shader);
181
182 GLint result = 0;
183 GLchar log[1024] = {0};
184
185 glGetShaderiv(current_shader, GL_COMPILE_STATUS, &result);
186 if (!result) {
187 glGetShaderInfoLog(current_shader, sizeof(log), NULL, log);
188 IGA_ERROR("Error compiling " + std::to_string(type) + " shader. ");//+std::to_string(log));
189 return;
190 }
191
192 glAttachShader(program, current_shader);
193 }

226 void create_framebuffer()
227 {
228 glGenFramebuffers(1, &FBO);
229 glBindFramebuffer(GL_FRAMEBUFFER, FBO);
230
231 glGenTextures(1, &texture_id);
232 glBindTexture(GL_TEXTURE_2D, texture_id);
233 glTexImage2D(GL_TEXTURE_2D, 0, GL_RGB, WIDTH, HEIGHT, 0, GL_RGB, GL_UNSIGNED_BYTE, NULL);
234 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
235 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
236 glFramebufferTexture2D(GL_FRAMEBUFFER, GL_COLOR_ATTACHMENT0, GL_TEXTURE_2D, texture_id, 0);
237
238 glGenRenderbuffers(1, &RBO);
239 glBindRenderbuffer(GL_RENDERBUFFER, RBO);
240 glRenderbufferStorage(GL_RENDERBUFFER, GL_DEPTH24_STENCIL8, WIDTH, HEIGHT);
241 glFramebufferRenderbuffer(GL_FRAMEBUFFER, GL_DEPTH_STENCIL_ATTACHMENT, GL_RENDERBUFFER, RBO);
242
243 if (glCheckFramebufferStatus(GL_FRAMEBUFFER) != GL_FRAMEBUFFER_COMPLETE)
244 IGA_ERROR("ERROR::FRAMEBUFFER:: Framebuffer is not complete!");
245
246 glBindFramebuffer(GL_FRAMEBUFFER, 0);
247 glBindTexture(GL_TEXTURE_2D, 0);
248 glBindRenderbuffer(GL_RENDERBUFFER, 0);
249 }
